<script>{`
  (function() {
    const filterBar = document.getElementById('filterBar');
    if (!filterBar) return;
    const buttons = filterBar.querySelectorAll('.filter-btn');
    const allCards = document.querySelectorAll('.video-card');
    const sections = document.querySelectorAll('#section-official, #section-community, #section-conference');
    const emptyState = document.getElementById('emptyState');
    const learningPath = document.querySelector('.learning-path');

    buttons.forEach(function(btn) {
      btn.addEventListener('click', function() {
        buttons.forEach(function(b) { b.classList.remove('active'); });
        btn.classList.add('active');
        var filter = btn.dataset.filter;
        var visibleCount = 0;

        if (filter === 'all') {
          allCards.forEach(function(card) { card.style.display = ''; });
          sections.forEach(function(sec) { sec.style.display = ''; });
          if (learningPath) learningPath.style.display = '';
          if (emptyState) emptyState.style.display = 'none';
          return;
        }

        if (learningPath) learningPath.style.display = 'none';

        allCards.forEach(function(card) {
          var cat = card.dataset.category;
          var level = card.dataset.level;
          var type = card.dataset.type;
          var match = false;

          if (filter === 'official' && cat === 'official') match = true;
          if (filter === 'community' && cat === 'community') match = true;
          if (filter === 'conference' && (cat === 'conference' || type === 'conference')) match = true;
          if (filter === 'tutorial' && type === 'tutorial') match = true;
          if (filter === 'beginner' && level === 'beginner') match = true;
          if (filter === 'intermediate' && level === 'intermediate') match = true;
          if (filter === 'advanced' && level === 'advanced') match = true;

          card.style.display = match ? '' : 'none';
          if (match) visibleCount++;
        });

        sections.forEach(function(sec) {
          var cards = sec.querySelectorAll('.video-card');
          var anyVisible = Array.from(cards).some(function(c) { return c.style.display !== 'none'; });
          sec.style.display = anyVisible ? '' : 'none';
        });

        if (emptyState) emptyState.style.display = visibleCount === 0 ? '' : 'none';
      });
    });
  })();
`}</script>
